迴圈式敘述 ( iteration ) 結構

迴圈的⼩幫⼿ break; vs. continue;

  • break;
    • ⽴刻停⽌整個迴圈的執⾏，然後執⾏迴圈之後的敘述
    • 直接跳離迴圈
    • 必須放在區塊內
  • continue;
    • 對於「for 迴圈」：先執⾏條件更新的部份，接著判斷條件以便決定是否要繼續執⾏該迴圈
    • 對於「while、do-while 迴圈」：直接判斷條件以便決定是否要繼續執⾏該迴圈
    • 必須放在迴圈之內，只有針對迴圈有⽤
    • 回到包住的那個迴圈的判斷式

package exer01;

import java.util.Scanner;

// 1. 輸⼊⼀些介於0-100之間的整數(按q離開)，找出最⼤值、最⼩值、總和、平均
// 2. 輸⼊⼀些整數(按q離開)，找出最⼤值、最⼩值、總和、平均

public class MaxMinNumber {
  public static void main(String[] args) {
    Scanner scan = new Scanner(System.in);

    // 1.
    // final int MAX = 100; // 有設定區間的：最⼤值就是區間最⼤數
    // final int MIN = 0; // 有設定區間的：最⼩值就是區間最⼩數

    // 2.
    final int MAX = Integer.MAX_VALUE;
    final int MIN = Integer.MIN_VALUE;

    int max = MIN; // 初值：最⼤值給最⼩數
    int min = MAX; // 初值：最⼩值給最⼤數

    int score = 0;
    int n = 0;
    int sum = 0;

    System.out.println("Enter the first score(q to quit)：");
    while(scan.hasNextInt()) {
      score = scan.nextInt();

      if(score < MIN || score > MAX) {
        System.out.println(score + "is an invalid value. Try again.");
        continue;
      }
      n++;
      sum += score;

      max = (score > max) ? score : max;
      min = (score < min) ? score : min;
      System.out.println("Enter next score(q to quit)：");
    }
    if(n != 0) {
      System.out.println("Average of " + n + " scores is " + (float)(sum) / n);
      System.out.println("The maximum is " + max + " and the minimum is " + min);
    }else {
      System.out.println("No valid scores were entered！");
    }
  }
}
  • 程式練習：輸⼊ 10 次⾼爾夫的成績，計算總桿數，平均桿數，以及⾼於或低於標準桿 ( PAR = 72 ) 幾桿
package exer01;

// 輸⼊ 10 次⾼爾夫的成績，計算總桿數，平均桿數，以及⾼於或低於標準桿 ( PAR = 72 ) 幾桿

import java.util.Scanner;

public class GolfScores {

  public static void main(String[] args) {
    Scanner scan = new Scanner(System.in);

    final int SIZE = 10;
    int[] score = new int [SIZE];

    final int PAR = 72; // 標準桿

    System.out.println("Enter 10 golf scores: ");
    for(int i = 0; i < SIZE; i++) { // 輸⼊ 10 次成績
      if(scan.hasNextInt()) {
      score[i] = scan.nextInt();
      }
    } // 以空⽩ ( space, tab, enter ) 當成輸⼊的結束

    for(int i = 0; i < SIZE; i++) { // 印出 10 次成績
      System.out.print(score[i] + " ");
    }
    System.out.println();

    int sum = 0;
    for(int i = 0; i < SIZE; i++) { // 加總 10 次成績
      sum += score[i];
    }

    float average = (float)sum / SIZE;

    System.out.println("Sum of scores = " + sum + ", average = " + average);
    System.out.println("That's a handicap of " + Math.round(average - PAR));
  }
}

public class Paint {

  public static void main(String[] args) {
    Scanner scan = new Scanner(System.in);

    final int COVERAGE = 200;

    int sq_feet = 0;
    int cans = 0;

    System.out.println("Enter number of square feet to be painted: (q to quit)");
    
    while(scan.hasNextInt()) {
      sq_feet = scan.nextInt();
      
      cans = sq_feet / COVERAGE ;
      cans += (sq_feet % COVERAGE == 0) ? 0 : 1;
      System.out.println("You need " + cans + (cans <= 1 ? " can " : " cans ") + "of paint.");
      System.out.println("Enter next value (q to quit) : ");
    }
    System.out.println("Done.");
  }
}
  • 程式練習：輸⼊⼀個正整數，計算該數的所有因數 ( 1 和本⾝以外 )，如果沒有因數就印出 " 該數是質數 "
package exer01;

// 輸⼊⼀個正整數，計算該數的所有因數 ( 1 和本⾝以外 )，如果沒有因數就印出 " 該數是質數 "

import java.util.Scanner;

public class Division {

  public static void main(String[] args) {
    Scanner scan = new Scanner(System.in);
    int num;
    int div;
    boolean isPrime;

    System.out.println("Enter an integer for analysis. Enter q to quit.");

    while(scan.hasNextInt()) {
      num = scan.nextInt();

      for(div = 2, isPrime = true; div * div <= num; div++) {
        if(num % div == 0) {
          if(div * div != num) {
            System.out.println(num + " is divisible by " + div + " and " + num / div);
          }else {
            System.out.println(num + " is divisible by " + div);
          }
          isPrime = false;
        }
      }
      if(isPrime == true) {
        System.out.println(num + " is prime.");
      }
    }
    System.out.println("Done.");
  }
}
